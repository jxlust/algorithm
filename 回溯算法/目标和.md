/**
 * @param {number[]} nums
 * @param {number} S
 * @return {number}
 */
var findTargetSumWays = function (nums, S) {
    let count = 0;
    let opt = [-1, 1];
    let res = S;
    let i = 0;
    var backtrack = function (res, i) {
        if (i === nums.length) {
            //遍历到了树的叶子节点，到底了
            if (res === 0) {
                count++;
            }
            return;
        }
        //选择列表
        for (let v of opt) {
            //做选择
            res -= v * nums[i];
            backtrack(res, i + 1);
            //撤销选择
            res += v * nums[i];
        }
    }
    backtrack(res, 0)
    return count;
};
