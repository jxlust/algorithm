### 这类博弈题目，思想是极小化极大


### 1. DFS加备忘录优化
```javascript
/**
 * @param {number[]} nums
 * @return {boolean}
 */
var PredictTheWinner = function (nums) {
    let n = nums.length;
    if (n % 2 === 0) return true;
    //奇数情况
    let map = new Map();
    /**
     * 返回当前选择人能选择的 当前选择的和下一个人选择的 差值最大值，也就是当前能选择的最优选择
     */
    const dfs = function (left, right) {
        if (left === right) {
            return nums[left]
        }
        let key = `${left}-${right}`;
        if(map.has(key)) return map.get(key);
        //选择左边
        let vL = nums[left] - dfs(left + 1, right)
        //选择右边
        let vR = nums[right] - dfs(left, right - 1)
        let M = Math.max(vL, vR);
        map.set(key, M)
        return M;
    }
    return dfs(0, n - 1) >= 0
};
```

### 2. 动态规划

```javascript
/**
 * @param {number[]} nums
 * @return {boolean}
 */
var PredictTheWinner = function (nums) {
    let n = nums.length;
    if (n % 2 === 0) return true;
    //奇数情况

    //dp[i][j] 表示i-j选择的最优结果
    let dp = new Array(n);
    for (let i = 0; i < n; i++) {
        dp[i] = new Array(n)
        dp[i][i] = nums[i]
    }

    // dp[0][n-1] 为第一个人能选择最大值结果
    // dp[i][j] = Math.max(nums[i] - dp[i+1][j],nums[j] - dp[i][j-1])

    //通过dp table很容易得出状态转移关系

    for (let i = n - 2; i >= 0; i--) {
        for (let j = i + 1; j < n; j++) {
            dp[i][j] = Math.max(nums[i] - dp[i + 1][j], nums[j] - dp[i][j - 1]);
        }
    }
    return dp[0][n - 1] >= 0

};
```
