### 题目
给出两个 非空 的链表用来表示两个非负的整数。其中，它们各自的位数是按照 逆序 的方式存储的，并且它们的每个节点只能存储 一位 数字。

如果，我们将这两个数相加起来，则会返回一个新的链表来表示它们的和。

您可以假设除了数字 0 之外，这两个数都不会以 0 开头。

### 代码
```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
var addTwoNumbers = function (l1, l2) {
    let key = 0;//保存进位数
    let result = new ListNode(-1);
    let p = result;
    let sum = 0;
    while (l1 !== null && l2 !== null) {
        sum = l1.val + l2.val;
        p.next = new ListNode((sum + key) % 10);
        //取进位
        key = ~~((sum + key) / 10);
        l1 = l1.next;
        l2 = l2.next;
        p = p.next;
    }

    //多余的要进行处理
    let remain = l1 !== null ? l1 : l2;
    while (remain !== null) {
        sum = remain.val;
        p.next = new ListNode( (sum + key) % 10 );
        key = ~~((sum + key) / 10);
        remain = remain.next;
        p = p.next;
    }
    if (key !== 0) {
        p.next = new ListNode(key)
    }
    return result.next;
};
```
### 优化代码
多余数据的处理逻辑跟上面的相似，可以合并处理
```javascript
var addTwoNumbers = function (l1, l2) {
    let key = 0;//保存进位数
    let result = new ListNode(-1);
    let p = result;
    let sum = 0;
    while (l1 || l2) {
        sum = 0;
        if (l1) {
            sum += l1.val
        }
        if (l2) {
            sum += l2.val
        }
        p.next = new ListNode((sum + key) % 10);
        //取进位
        key = ~~((sum + key) / 10);
        if (l1) {
            l1 = l1.next;
        }
        if (l2) {
            l2 = l2.next;
        }
        p = p.next;
    }

    if (key !== 0) {
        p.next = new ListNode(key)
    }
    return result.next;
};
```
