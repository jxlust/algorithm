## 二叉堆的数据结构
在js里面没有这种数据结构可以直接使用，不像C++或java里面直接有优先队列priority_queue是已堆的形式维护的

## 代码实现

```javascript
/**
 * 默认建堆方式是建立最大堆
 */
class HeapIfy {
  constructor(arr) {
    this.arr = arr;
    let n = arr.length;
    //从底往上进行堆调整
    for (let i = n >> 1; i >= 0; i--) {
      this.maxHeapIfy(i, n)
    }
  }
  /**
   * 插入元素
   */
  push(value) {
    this.arr.unshift(value);
    //继续堆调整
    this.maxHeapIfy(0, this.arr.length)
  }
  /**
   * 获取堆顶元素
   */
  top() {
    return this.arr[0]
  }
  /**替换掉top */
  replaceTop(v) {
    this.arr[0] = v;
    //继续堆调整
    this.maxHeapIfy(0, this.arr.length)
  }
  swap(arr, left, right) {
    // [arr[left], arr[right]] = [arr[right], arr[left]]
    let temp = arr[left];
    arr[left] = arr[right];
    arr[right] = temp;
  }
  /**
   * 弹出堆顶元素
   */
  shift() {
    this.arr.shift()
    //继续堆调整
    this.maxHeapIfy(0, this.arr.length)
  }
  maxHeapIfy(index, len) {
    let maxIndex = index;
    let left = 2 * index + 1,
      right = 2 * index + 2;
    if (left < len && this.arr[left] > this.arr[maxIndex]) {
      maxIndex = left;
    }
    if (right < len && this.arr[right] > this.arr[maxIndex]) {
      maxIndex = right;
    }
    if (maxIndex != index) {
      //交互位置
      this.swap(this.arr, maxIndex, index);
      //因为调整了最大值，可能破坏了maxIndex子树的最大堆的结构,继续堆调整
      this.maxHeapIfy(maxIndex, len)
    }
  }
}
```
排序和topK方法测试代码：
```javascript
let test = [8, 2, 99, 4, 32, 99, 34, 76, 22];
console.log(getminK(test, 3));
let heap = new HeapIfy(test);
console.log(1, test); // [ 99, 76, 99, 22, 32, 8, 34, 4, 2 ]
//插入
heap.push(88);
console.log(2, test); // [ 99, 99, 76, 88, 22, 32, 8, 34, 4, 2 ]

heap.shift();
console.log(3, test); //[ 99, 76, 88, 22, 32, 8, 34, 4, 2 ]
console.log('top:', heap.top()); // top: 99

//下面是进行排序
function sort(heap, test) {
  for (let i = test.length - 1; i >= 0; i--) {
    heap.swap(test, i, 0);
    heap.maxHeapIfy(0, i);
  }
}

function getminK(arr, k) {
  let mink = arr.slice(0, k);
  let heap = new HeapIfy(mink);
  for (let i = k; i < arr.length; i++) {
    let top = heap.top();
    if (top > arr[i]) {
      //说明更小的arr[i]得入堆
      heap.replaceTop(arr[i])
    }
  }
  return mink;
}
```
